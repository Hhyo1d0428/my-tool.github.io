        <label>1. 합칠 EPUB 파일들 추가 (순서 무관)</label>
        <input type="file" id="epubInput" accept=".epub" multiple>
        
        <div id="fileListContainer"></div>

        <div id="metaSection" style="display:none; margin-top: 25px;">
            <label>2. 파일 저장 설정 및 메타데이터</label>
            <input type="text" id="fileName" placeholder="저장될 파일 이름 (예: 합본최종.epub)">
            <input type="text" id="bookTitle" placeholder="뷰어에 보일 책 제목 (예: 내가 키운 S급 완전판)">
            <input type="text" id="bookAuthor" placeholder="저자 이름">
            
            <button onclick="processMerge()" class="btn-green">합본 분석 및 목차 생성 (1단계)</button>

        <div id="tocSection" style="display:none; margin-top: 25px; border-top: 2px dashed #ccc; padding-top: 20px;">
            <label>3. 최종 목차 확인 및 정리</label>
            <p style="font-size: 13px; color: gray; margin-top: 5px;">합쳐질 책의 목차입니다. 이름을 수정하거나 불필요한 목차(표지, 시작 등)는 ❌ 버튼으로 지워주세요.</p>
            <div id="tocList"></div>
            
            <button onclick="downloadMergedEPUB()">최종 합본 다운로드 (2단계)</button>

    <script>
        let selectedFiles = [];
        let mergedZip = new JSZip();
        let finalTOC = [];
        
        const epubInput = document.getElementById('epubInput');
        const fileListContainer = document.getElementById('fileListContainer');
        const metaSection = document.getElementById('metaSection');
        
        function resolvePath(basePath, relativePath) {
            let parts = basePath.split('/');
            parts.pop(); 
            let relParts = relativePath.split('/');
            for (let p of relParts) {
                if (p === '.') continue;
                if (p === '..') parts.pop();
                else parts.push(p);
            }
            return parts.join('/');
        }

        epubInput.addEventListener('change', function(e) {
            for(let file of e.target.files) {
                selectedFiles.push(file);
            }
            renderFileList();
            epubInput.value = ""; 
        });

        function renderFileList() {
            fileListContainer.innerHTML = "";
            if(selectedFiles.length > 0) metaSection.style.display = "block";
            else metaSection.style.display = "none";
            
            selectedFiles.forEach((file, index) => {
                let div = document.createElement('div');
                div.className = 'file-item';
                div.innerHTML = `
                    <div style="font-weight:bold; word-break:break-all;">${index + 1}. ${file.name}</div>
                    <div class="file-controls">
                        <button class="btn-move" onclick="moveFile(${index}, -1)">⬆️ 위로</button>
                        <button class="btn-move" onclick="moveFile(${index}, 1)">⬇️ 아래로</button>
                        <button class="btn-red" onclick="removeFile(${index})">❌ 제거</button>
                    </div>
                `;
                fileListContainer.appendChild(div);
            });
        }

        function moveFile(index, dir) {
            if (index + dir < 0 || index + dir >= selectedFiles.length) return;
            let temp = selectedFiles[index];
            selectedFiles[index] = selectedFiles[index + dir];
            selectedFiles[index + dir] = temp;
            renderFileList();
        }

        function removeFile(index) {
            selectedFiles.splice(index, 1);
            renderFileList();
        }

        async function processMerge() {
            if (selectedFiles.length < 1) return alert("파일을 추가해주세요.");
            
            const status = document.getElementById('processStatus');
            status.innerText = "⏳ 파일들을 하나로 포장 중입니다... (10~20초 소요될 수 있습니다)";
            status.className = "status";
            document.getElementById('tocSection').style.display = "none";
            
            mergedZip = new JSZip();
            mergedZip.file("mimetype", "application/epub+zip", { compression: "STORE" });
            
            let combinedManifest = "";
            let combinedSpine = "";
            finalTOC = [];
            let coverId = "";
            
            try {
                for (let i = 0; i < selectedFiles.length; i++) {
                    let fileZip = await JSZip.loadAsync(selectedFiles[i]);
                    
                    let containerXml = await fileZip.file("META-INF/container.xml").async("string");
                    let containerDoc = new DOMParser().parseFromString(containerXml, "text/xml");
                    let opfPath = containerDoc.getElementsByTagName("rootfile")[0].getAttribute("full-path");
                    let opfContent = await fileZip.file(opfPath).async("string");
                    let opfDoc = new DOMParser().parseFromString(opfContent, "text/xml");
                    
                    if (i === 0) {
                        let titleTag = opfDoc.getElementsByTagName("dc:title")[0];
                        let authorTag = opfDoc.getElementsByTagName("dc:creator")[0];
                        
                        let tempTitle = titleTag ? titleTag.textContent + " (합본)" : "합본";
                        
                        if(!document.getElementById('bookTitle').value) document.getElementById('bookTitle').value = tempTitle;
                        if(!document.getElementById('fileName').value) document.getElementById('fileName').value = tempTitle + ".epub";
                        if(authorTag && !document.getElementById('bookAuthor').value) document.getElementById('bookAuthor').value = authorTag.textContent;
                        
                        let metaTags = opfDoc.getElementsByTagName("meta");
                        for (let meta of metaTags) {
                            if (meta.getAttribute("name") === "cover") coverId = "b0_" + meta.getAttribute("content");
                        }
                    }

                    let items = opfDoc.getElementsByTagName("item");
                    for (let item of items) {
                        let id = item.getAttribute("id");
                        let href = item.getAttribute("href");
                        let mediaType = item.getAttribute("media-type");
                        let properties = item.getAttribute("properties");
                        
                        // Pages 앱 등이 만든 기존 EPUB3 지도의 권한 박탈 (새 지도로 대체하기 위함)
                        if (properties && properties.includes("nav")) {
                            properties = properties.replace(/\bnav\b/g, "").trim();
                        }
                        
                        let absolutePath = resolvePath(opfPath, href);
                        let fileData = await fileZip.file(absolutePath).async("arraybuffer");
                        
                        let newPath = `book${i}/${absolutePath}`;
                        mergedZip.file(newPath, fileData);
                        
                        let propAttr = properties ? ` properties="${properties}"` : "";
                        if (i === 0 && (properties.includes("cover-image") || id === coverId.replace("b0_", ""))) {
                            coverId = `b${i}_${id}`;
                        }
                        combinedManifest += `<item id="b${i}_${id}" href="${newPath}" media-type="${mediaType}"${propAttr}/>\n`;
                    }
                    
                    let itemrefs = opfDoc.getElementsByTagName("itemref");
                    for (let ref of itemrefs) {
                        combinedSpine += `<itemref idref="b${i}_${ref.getAttribute("idref")}"/>\n`;
                    }

                    let ncxPath = Object.keys(fileZip.files).find(k => k.endsWith('.ncx'));
                    if (ncxPath) {
                        let ncxContent = await fileZip.file(ncxPath).async("string");
                        let navPoints = new DOMParser().parseFromString(ncxContent, "text/xml").getElementsByTagName("navPoint");
                        
                        for (let nav of navPoints) {
                            let text = nav.getElementsByTagName("text")[0].textContent.trim();
                            let src = nav.getElementsByTagName("content")[0].getAttribute("src");
                            let absSrc = resolvePath(ncxPath, src);
                            finalTOC.push({ id: Math.random().toString(36).substr(2, 9), title: text, src: `book${i}/${absSrc}` });
                        }
                    }
                }

                let finalTitle = document.getElementById('bookTitle').value || "합본";
                let finalAuthor = document.getElementById('bookAuthor').value || "작자미상";
                let uuid = "urn:uuid:" + Date.now();
                
                // 마스터 OPF에 구형 지도(ncx)와 최신 지도(nav.xhtml)를 동시에 심어줌
                let masterOPF = `<?xml version="1.0" encoding="utf-8"?>
<package xmlns="http://www.idpf.org/2007/opf" version="3.0" unique-identifier="pub-id">
    <metadata xmlns:dc="http://purl.org/dc/elements/1.1/">
        <dc:title>${finalTitle}</dc:title>
        <dc:creator>${finalAuthor}</dc:creator>
        <dc:identifier id="pub-id">${uuid}</dc:identifier>
        <dc:language>ko</dc:language>
        ${coverId ? `<meta name="cover" content="${coverId}"/>` : ""}
    </metadata>
    <manifest>
        <item id="master_ncx" href="toc.ncx" media-type="application/x-dtbncx+xml"/>
        <item id="master_nav" href="nav.xhtml" media-type="application/xhtml+xml" properties="nav"/>
        ${combinedManifest}
    </manifest>
    <spine toc="master_ncx">
        ${combinedSpine}
    </spine>
</package>`;

                mergedZip.file("content.opf", masterOPF);
                mergedZip.file("META-INF/container.xml", `<?xml version="1.0"?><container version="1.0" xmlns="urn:oasis:names:tc:opendocument:xmlns:container"><rootfiles><rootfile full-path="content.opf" media-type="application/oebps-package+xml"/></rootfiles></container>`);

                renderTOC();
                status.innerText = "✅ 병합 및 목차 추출 완료! 불필요한 목차를 정리해주세요.";
                status.className = "status success";
                document.getElementById('tocSection').style.display = "block";

            } catch (err) {
                status.innerText = "❌ 오류: " + err.message;
                status.className = "status error";
            }
        }

        function removeTOCItem(id) {
            finalTOC = finalTOC.filter(item => item.id !== id);
            renderTOC();
        }

        function renderTOC() {
            const list = document.getElementById('tocList');
            list.innerHTML = "";
            finalTOC.forEach((item, idx) => {
                let div = document.createElement('div');
                div.className = "toc-item";
                div.innerHTML = `
                    <span style="color:#666; font-size: 14px; width: 25px;">${idx+1}.</span>
                    <input type="text" id="toc_input_${item.id}" value="${item.title}">
                    <button class="btn-red toc-del-btn" onclick="removeTOCItem('${item.id}')">❌</button>
                `;
                list.appendChild(div);
            });
        }

        async function downloadMergedEPUB() {
            const status = document.getElementById('downloadStatus');
            status.innerText = "⏳ 최종 파일 생성 중...";
            status.className = "status";

            let ncxNavPoints = "";
            let htmlNavLis = "";
            
            finalTOC.forEach((item, idx) => {
                let userTitle = document.getElementById(`toc_input_${item.id}`).value;
                
                // 1. 구형 지도(NCX) 데이터 생성
                ncxNavPoints += `
                <navPoint id="navPoint-${idx+1}" playOrder="${idx+1}">
                    <navLabel><text>${userTitle}</text></navLabel>
                    <content src="${item.src}"/>
                </navPoint>`;
                
                // 2. 신형 지도(EPUB3 nav.xhtml) 데이터 생성
                htmlNavLis += `<li><a href="${item.src}">${userTitle}</a></li>\n`;
            });

            let finalTitle = document.getElementById('bookTitle').value || "합본";
            
            // 구형 지도 파일 생성
            let masterNCX = `<?xml version="1.0" encoding="UTF-8"?>
<ncx xmlns="http://www.daisy.org/z3998/2012/vocab/structure/#" version="2005-1">
    <head><meta name="dtb:uid" content="urn:uuid:${Date.now()}"/></head>
    <docTitle><text>${finalTitle}</text></docTitle>
    <navMap>${ncxNavPoints}</navMap>
</ncx>`;
            mergedZip.file("toc.ncx", masterNCX);

            // 신형 지도 파일 생성
            let masterNavHtml = `<?xml version="1.0" encoding="utf-8"?>
<html xmlns="http://www.w3.org/1999/xhtml" xmlns:epub="http://www.idpf.org/2007/ops">
<head><title>${finalTitle}</title></head>
<body>
    <nav epub:type="toc" id="toc">
        <h1>목차</h1>
        <ol>
            ${htmlNavLis}
        </ol>
    </nav>
</body>
</html>`;
            mergedZip.file("nav.xhtml", masterNavHtml);

            try {
                const blob = await mergedZip.generateAsync({ type: "blob", compression: "DEFLATE", compressionOptions: { level: 6 } });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                
                let customFileName = document.getElementById('fileName').value.trim();
                if(!customFileName) customFileName = finalTitle + ".epub";
                if(!customFileName.toLowerCase().endsWith('.epub')) customFileName += ".epub";
                
                a.download = customFileName;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                
                status.innerText = "✅ 다운로드 성공! (도서 앱에서 새 책으로 열립니다)";
                status.className = "status success";
            } catch(err) {
                status.innerText = "❌ 다운로드 오류: " + err.message;
                status.className = "status error";
            }
        }
    </script>
